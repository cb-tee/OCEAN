    window.onload = function() {

        // ดึงข้อมูลส่วนตัว
        const name = localStorage.getItem('user_name') || "ไม่ระบุชื่อ";
        const gender = localStorage.getItem('user_gender') || "ไม่ระบุ";
        const status = localStorage.getItem('user_status') || "ไม่ระบุ";

        // ดึงคะแนนที่แอบบันทึกไว้จากหน้าต่างๆ มาเตรียมไว้
        const p1 = localStorage.getItem('p1') || 0;
        const p2 = localStorage.getItem('p2') || 0;
        const p3 = localStorage.getItem('p3') || 0;
        const p4 = localStorage.getItem('p4') || 0;
        const p5 = localStorage.getItem('p5') || 0;

        // เตรียมข้อมูลส่งไป n8n (จะไม่มีการแสดงผลบนหน้าจอ)
        const payload = {
            username: name,    // เปลี่ยนจาก "Anonymous_User" เป็นตัวแปร name
            gender: gender,    // เพิ่มบรรทัดนี้เพื่อส่งเพศ
            status: status,    // เพิ่มบรรทัดนี้เพื่อส่งสถานะ 
            score_openness: p1,
            score_conscientiousness: p2,
            score_extraversion: p3,
            score_agreeableness: p4,
            score_neuroticism: p5
        };

        // ยิงข้อมูลเข้า n8n Webhook
        fetch('https://cb-tee.app.n8n.cloud/webhook/OCEAN-survey-result', { // ใช้ URL ทดสอบของคุณ
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload)
        })
        .then(response => {
            console.log("Data sent successfully!");
            // ถ้าส่งเสร็จแล้วอยากให้ล้างคะแนนในเครื่องทิ้งเพื่อเริ่มใหม่
            // localStorage.clear(); 
        })
        .catch(err => console.error("Error:", err));
    };
</script>
